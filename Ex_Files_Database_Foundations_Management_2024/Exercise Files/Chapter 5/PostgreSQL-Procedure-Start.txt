-- Create a custom procedure in PostgreSQL-- Create the table to hold values for this exampleCREATE TEMPORARY TABLE my_names (my_name varchar(50));-- Create a stored procedure to aid in inserting rows into the tableCREATE PROCEDURE insert_name (first_name varchar(50))LANGUAGE SQLAS $$INSERT INTO my_names VALUES (first_name);$$;-- Execute the procedure and pass it a parameter valueCALL insert_name('Amos');-- Verify the resultsSELECT * FROM my_names;-- Query a system catalog to view all procedures on the databaseSELECT * FROM pg_procWHERE prokind = 'p';-- Clean up the databaseDROP TABLE my_names;DROP PROCEDURE insert_name;
